# 堆
## 堆是一颗完全二叉树，以大根堆为例，顺序存储从1开始，堆顶是最大值
堆排序：
1. 建堆，倒着将每个结点为根的子树调整为堆（最后一个非叶节点开始），并且需要依次向下调整
2. 排序，每轮堆顶换到最后，向下调整新的堆顶（原堆顶到末尾，不再考虑），总共n - 1轮，n为结点数
3. 堆排序不稳定
```c++
class MinHeap {
private:
    vector<int> heap;
    
    // 获取父节点索引
    int parent(int i) { return (i - 1) / 2; }
    
    // 获取左子节点索引
    int leftChild(int i) { return 2 * i + 1; }
    
    // 获取右子节点索引
    int rightChild(int i) { return 2 * i + 2; }
    
    // 向上堆化（用于插入后调整）
    void siftUp(int i) {
        while (i > 0 && heap[parent(i)] > heap[i]) {
            swap(heap[i], heap[parent(i)]);
            i = parent(i);
        }
    }
    
    // 向下堆化（用于删除后调整）
    void siftDown(int i) {
        int minIndex = i;
        int size = heap.size();
        
        int l = leftChild(i);
        if (l < size && heap[l] < heap[minIndex]) {
            minIndex = l;
        }
        
        int r = rightChild(i);
        if (r < size && heap[r] < heap[minIndex]) {
            minIndex = r;
        }
        
        if (i != minIndex) {
            swap(heap[i], heap[minIndex]);
            siftDown(minIndex);
        }
    }

public:
    // 默认构造函数
    MinHeap() {}
    
    // 从数组构建堆
    MinHeap(const vector<int>& array) {
        heap = array;
        // 从最后一个非叶子节点开始堆化
        for (int i = (heap.size() - 2) / 2; i >= 0; i--) {
            siftDown(i);
        }
    }
    
    // 插入元素
    void insert(int value) {
        heap.push_back(value);
        siftUp(heap.size() - 1);
    }
    
    // 获取最小值
    int getMin() {
        if (heap.empty()) {
            throw out_of_range("Heap is empty");
        }
        return heap[0];
    }
    
    // 删除最小值
    int extractMin() {
        if (heap.empty()) {
            throw out_of_range("Heap is empty");
        }
        
        int min = heap[0];
        heap[0] = heap.back();
        heap.pop_back();
        
        if (!heap.empty()) {
            siftDown(0);
        }
        
        return min;
    }
    
    // 更新值
    void update(int i, int newValue) {
        if (i < 0 || i >= heap.size()) {
            throw out_of_range("Index out of range");
        }
        
        int oldValue = heap[i];
        heap[i] = newValue;
        
        if (newValue < oldValue) {
            siftUp(i);
        } else {
            siftDown(i);
        }
    }
    
    // 删除任意元素
    void remove(int i) {
        if (i < 0 || i >= heap.size()) {
            throw out_of_range("Index out of range");
        }
        
        // 将该元素设为最小值（负无穷）
        heap[i] = INT_MIN;
        siftUp(i);
        
        // 删除最小值
        extractMin();
    }
    
    // 堆排序（升序）
    vector<int> heapSort() {
        vector<int> result;
        vector<int> temp = heap;  // 保存原始堆
        
        while (!heap.empty()) {
            result.push_back(extractMin());
        }
        
        heap = temp;  // 恢复原始堆
        return result;
    }
    
    // 检查堆属性
    bool isHeap() {
        int size = heap.size();
        for (int i = 0; i <= (size - 2) / 2; i++) {
            if (leftChild(i) < size && heap[i] > heap[leftChild(i)]) {
                return false;
            }
            if (rightChild(i) < size && heap[i] > heap[rightChild(i)]) {
                return false;
            }
        }
        return true;
    }
    
    // 获取堆大小
    int size() {
        return heap.size();
    }
    
    // 检查堆是否为空
    bool isEmpty() {
        return heap.empty();
    }
    
    // 打印堆
    void print() {
        cout << "Heap: [";
        for (int i = 0; i < heap.size(); i++) {
            cout << heap[i];
            if (i < heap.size() - 1) cout << ", ";
        }
        cout << "]" << endl;
    }
    
    // 可视化打印堆
    void printTree() {
        if (heap.empty()) {
            cout << "Heap is empty" << endl;
            return;
        }
        
        // 计算树的高度
        int height = 0;
        int nodes = 1;
        while (nodes <= heap.size()) {
            height++;
            nodes *= 2;
        }
        
        // 按层级打印
        int levelStart = 0;
        int levelSize = 1;
        
        for (int h = 0; h < height; h++) {
            // 打印缩进
            int indent = (1 << (height - h - 1)) - 1;
            for (int i = 0; i < indent; i++) {
                cout << " ";
            }
            
            // 打印当前层级节点
            for (int i = 0; i < levelSize; i++) {
                int index = levelStart + i;
                if (index < heap.size()) {
                    cout << heap[index];
                } else {
                    cout << " ";
                }
                
                // 打印节点间距
                int spacing = (1 << (height - h)) - 1;
                for (int s = 0; s < spacing; s++) {
                    cout << " ";
                }
            }
            cout << endl;
            
            // 打印连接线
            if (h < height - 1) {
                for (int i = 0; i < indent - 1; i++) {
                    cout << " ";
                }
                
                for (int i = 0; i < levelSize; i++) {
                    int index = levelStart + i;
                    if (index < heap.size()) {
                        if (leftChild(index) < heap.size()) {
                            cout << "/";
                        } else {
                            cout << " ";
                        }
                        
                        int spacing = (1 << (height - h)) - 3;
                        for (int s = 0; s < spacing; s++) {
                            cout << " ";
                        }
                        
                        if (rightChild(index) < heap.size()) {
                            cout << "\\";
                        } else {
                            cout << " ";
                        }
                        
                        for (int s = 0; s < (1 << (height - h)) - 1; s++) {
                            cout << " ";
                        }
                    }
                }
                cout << endl;
            }
            
            levelStart += levelSize;
            levelSize *= 2;
        }
    }
};
```
