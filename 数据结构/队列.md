```c++
#include <iostream>
#include <vector>
#include <stdexcept>

class MyQueue {
private:
    std::vector<int> data;  // 使用vector存储队列元素
    int front;              // 队头指针
    int rear;               // 队尾指针

public:
    // 构造函数
    MyQueue() : front(0), rear(0) {}

    // 检查队列是否为空
    bool isEmpty() {
        return front == rear;
    }

    // 获取队列大小
    int size() {
        return rear - front;
    }

    // 添加元素到队尾（增）
    void enqueue(int value) {
        data.push_back(value);
        rear++;
    }

    // 删除队头元素（删）
    void dequeue() {
        if (isEmpty()) {
            throw std::out_of_range("Queue underflow: Cannot dequeue from an empty queue");
        }
        front++;
    }

    // 获取队头元素（查）
    int peek() {
        if (isEmpty()) {
            throw std::out_of_range("Queue is empty: Cannot peek");
        }
        return data[front];
    }

    // 获取队尾元素（查）
    int peekRear() {
        if (isEmpty()) {
            throw std::out_of_range("Queue is empty: Cannot peek at rear");
        }
        return data[rear - 1];
    }

    // 修改队头元素（改）
    void updateFront(int value) {
        if (isEmpty()) {
            throw std::out_of_range("Queue is empty: Cannot update front element");
        }
        data[front] = value;
    }

    // 修改队尾元素（改）
    void updateRear(int value) {
        if (isEmpty()) {
            throw std::out_of_range("Queue is empty: Cannot update rear element");
        }
        data[rear - 1] = value;
    }

    // 修改指定位置的元素（改）
    void updateAt(int index, int value) {
        if (index < 0 || index >= size()) {
            throw std::out_of_range("Invalid index: Out of queue range");
        }
        data[front + index] = value;
    }

    // 查找元素是否在队列中（查）
    bool contains(int value) {
        for (int i = front; i < rear; i++) {
            if (data[i] == value) {
                return true;
            }
        }
        return false;
    }

    // 查找元素位置（查）
    int find(int value) {
        for (int i = front; i < rear; i++) {
            if (data[i] == value) {
                return i - front;  // 返回相对位置
            }
        }
        return -1;  // 未找到返回-1
    }

    // 打印队列内容（从队头到队尾）
    void print() {
        if (isEmpty()) {
            std::cout << "Queue is empty" << std::endl;
            return;
        }
        
        std::cout << "Front -> ";
        for (int i = front; i < rear; i++) {
            std::cout << data[i];
            if (i < rear - 1) std::cout << " -> ";
        }
        std::cout << " <- Rear" << std::endl;
    }
    
    // 内存优化：移除已出队元素（可选）
    void optimizeMemory() {
        if (front > 0) {
            // 复制剩余元素到新vector
            std::vector<int> newData(data.begin() + front, data.end());
            data = newData;
            rear = data.size();
            front = 0;
        }
    }
};

int main() {
    MyQueue queue;
    
    try {
        // 测试队列操作
        std::cout << "Enqueuing elements..." << std::endl;
        queue.enqueue(10);
        queue.enqueue(20);
        queue.enqueue(30);
        queue.print();  // 10 -> 20 -> 30
        
        std::cout << "\nFront element: " << queue.peek() << std::endl;  // 10
        std::cout << "Rear element: " << queue.peekRear() << std::endl; // 30
        
        std::cout << "\nUpdating front element to 15..." << std::endl;
        queue.updateFront(15);
        queue.print();  // 15 -> 20 -> 30
        
        std::cout << "\nUpdating rear element to 35..." << std::endl;
        queue.updateRear(35);
        queue.print();  // 15 -> 20 -> 35
        
        std::cout << "\nDequeuing element..." << std::endl;
        queue.dequeue();
        queue.print();  // 20 -> 35
        
        std::cout << "\nEnqueuing 40 and 50..." << std::endl;
        queue.enqueue(40);
        queue.enqueue(50);
        queue.print();  // 20 -> 35 -> 40 -> 50
        
        std::cout << "\nUpdating element at position 2 to 45..." << std::endl;
        queue.updateAt(2, 45); // 位置2是40（索引从0开始）
        queue.print();  // 20 -> 35 -> 45 -> 50
        
        std::cout << "\nIs 35 in the queue? " << (queue.contains(35) ? "Yes" : "No") << std::endl;
        std::cout << "Position of 45: " << queue.find(45) << std::endl;
        std::cout << "Position of 99: " << queue.find(99) << std::endl;
        
        std::cout << "\nCurrent queue size: " << queue.size() << std::endl;
        std::cout << "Is queue empty? " << (queue.isEmpty() ? "Yes" : "No") << std::endl;
        
        std::cout << "\nDequeuing all elements..." << std::endl;
        while (!queue.isEmpty()) {
            std::cout << "Dequeuing: " << queue.peek() << std::endl;
            queue.dequeue();
        }
        queue.print();  // Queue is empty
        
        // 测试异常处理
        std::cout << "\nTrying to dequeue from empty queue..." << std::endl;
        queue.dequeue();  // 将抛出异常
    } 
    catch (const std::exception& e) {
        std::cerr << "Error: " << e.what() << std::endl;
    }
    
    return 0;
}
```
